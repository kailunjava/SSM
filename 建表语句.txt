create table DIC(
ID number(11) primary key,
CODE varchar2(20)not null,
NAME varchar2(20) not null,
PID varchar2(11) not null
);

CREATE SEQUENCE DIC_sequence
INCREMENT BY 1 
START WITH 1 
NOMAXVALUE 
NOCYCLE 
NOCACHE 

CREATE trigger DIC_trig before
insert on  DIC for each row when (new.ID is null)
begin
select DIC_sequence.nextval into:new.ID from dual;
end;
